Q5. Write the definition of a method printPrime(NUMBERS) to print prime numbers of the list of NUMBERS.

def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def printPrime(NUMBERS):
    primes = [num for num in NUMBERS if is_prime(num)]
    print(", ".join(map(str, primes)))

Q6. Differentiate direct and indirect recursion with an example.

  • Direct Recursion: A function calls itself.
    def direct_recursion(n):
    if n > 0:
        print(n)
        direct_recursion(n-1)
  • Indirect Recursion: A function calls another function, which in turn calls the first function.
    def indirect_recursion1(n):
    if n > 0:
        print(n)
        indirect_recursion2(n-1)

def indirect_recursion2(n):
    if n > 0:
        indirect_recursion1(n-1)

Q7. Write a function isPrime(N) which inputs a number as an argument and returns 0 if the number is prime, else 1. Code another function isPalin(N) which inputs a number and returns 0 if the number is palindrome, else 1. Now in the main program, accept lower and upper range from the user to print palprime numbers between them.

def isPrime(N):
    if N <= 1:
        return 1
    for i in range(2, int(N**0.5) + 1):
        if N % i == 0:
            return 1
    return 0

def isPalin(N):
    return 0 if str(N) == str(N)[::-1] else 1

lower = int(input("Enter lower limit: "))
upper = int(input("Enter upper limit: "))
palprimes = []
for num in range(lower, upper + 1):
    if isPrime(num) == 0 and isPalin(num) == 0:
        palprimes.append(num)
print("The palprimes are:", " ".join(map(str, palprimes)))

Q1. Write a function findGrade(marks) which accepts marks as an argument and returns the grade as per the given criteria. Now in the main program, accept marks of 40 students in a subject of your class and predict the number of students in each grade.

def findGrade(marks):
    if 90 <= marks <= 100:
        return 'A'
    elif 75 <= marks < 90:
        return 'B'
    elif 60 <= marks < 75:
        return 'C'
    elif 45 <= marks < 60:
        return 'D'
    elif 33 <= marks < 45:
        return 'E'
    else:
        return 'F'

grades = {'A': 0, 'B': 0, 'C': 0, 'D': 0, 'E': 0, 'F': 0}
for _ in range(40):
    marks = int(input("Enter marks: "))
    grade = findGrade(marks)
    grades[grade] += 1

for grade, count in grades.items():
    print(f"Grade {grade}: {count} students")

Q2. Generate twin prime numbers between 1 and 100. Twin prime numbers are the two numbers whose difference is 2 and both are prime numbers.

def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

twins = []
for p in range(1, 100):
    if is_prime(p) and is_prime(p+2):
        twins.append((p, p+2))
print(twins)

    return result
